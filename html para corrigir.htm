        function formatDate(dateString) {

            if (!dateString) {

                return 'N/A';

            }

            const parts = String(dateString).split('-');

            if (parts.length === 3 && !isNaN(parseInt(parts[0])) && !isNaN(parseInt(parts[1])) && !isNaN(parseInt(parts[2]))) {

                return `${parts[2]}/${parts[1]}/${parts[0]}`;

            }

            return dateString;

        }
